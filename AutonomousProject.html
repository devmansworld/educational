# Simple 1D Kalman Filter Example
import numpy as np

class KalmanFilter:
    def __init__(self, process_var, meas_var, estimate, error_est):
        self.process_var = process_var
        self.meas_var = meas_var
        self.estimate = estimate
        self.error_est = error_est

    def update(self, measurement):
        # Prediction
        pred_est = self.estimate
        pred_err = self.error_est + self.process_var

        # Update
        gain = pred_err / (pred_err + self.meas_var)
        self.estimate = pred_est + gain * (measurement - pred_est)
        self.error_est = (1 - gain) * pred_err
        return self.estimate

import random, math

class ParticleFilter:
    def __init__(self, num_particles=50):
        self.particles = [random.random() for _ in range(num_particles)]
        self.weights = [1.0 / num_particles] * num_particles

    def update_weights(self, measurement, sigma):
        for i, particle in enumerate(self.particles):
            prob = math.exp(-0.5 * ((measurement - particle) / sigma)**2)
            self.weights[i] = prob
        total = sum(self.weights)
        self.weights = [w / total for w in self.weights]

import math

def displacement_components(D, angle_deg):
    angle_rad = math.radians(angle_deg)
    return D * math.cos(angle_rad), D * math.sin(angle_rad)

# Example usage
if __name__ == "__main__":
    x, y = displacement_components(1920, 30)
    print(f"x = {x:.2f}, y = {y:.2f}")
